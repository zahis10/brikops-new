+    title: str = Field(..., min_length=2, max_length=200)

+class SafetyTaskUpdate(BaseModel):
+    title: Optional[str] = None
+    severity: Optional[SafetySeverity] = None
+    description: Optional[str] = None
+    assignee_id: Optional[str] = None
+    company_id: Optional[str] = None
+    due_at: Optional[str] = None
+    status: Optional[SafetyTaskStatus] = None
+    corrective_action: Optional[str] = None
+    verification_photo_urls: Optional[List[str]] = None
+
+
+def _assert_transition_allowed(
+    current: str, next_status: str, project_role: str, payload_dict: dict
+) -> None:
+    """Enforce status transition rules. Raises 409 on invalid transition."""
+    if current == next_status:
+        return  # idempotent
+    # Spec: open → in_progress → completed; any → cancelled; reverse = 409
+    allowed = {
+        "open": {"in_progress", "cancelled"},
+        "in_progress": {"completed", "cancelled"},
+        "completed": {"cancelled"},
+        "cancelled": set(),
+    }
+    if next_status not in allowed.get(current, set()):
+        raise HTTPException(status_code=409, detail={
+            "current_status": current,
+            "requested": next_status,
+            "reason": "transition not allowed",
+        })
+    if next_status == "completed":
+        ca = payload_dict.get("corrective_action")
+        if not ca or not str(ca).strip():
+            raise HTTPException(status_code=409, detail={
+                "current_status": current,
+                "requested": next_status,
+                "reason": "corrective_action is required to mark a task completed",
+            })
+    if next_status == "cancelled":
+        if project_role != "project_manager":
+            raise HTTPException(status_code=409, detail={
+                "current_status": current,
+                "requested": next_status,
+                "reason": "only project_manager may cancel a task",
+            })
+

    64	def _hash_id_number(id_number: Optional[str]) -> Optional[str]:
    65	    """SHA-256 of stripped Israeli ID / passport. Lookup aid; never returned to client."""
    66	    if not id_number:
    67	        return None
    68	    return hashlib.sha256(id_number.strip().encode("utf-8")).hexdigest()

    71	def _retention_date(from_iso: str) -> str:
    72	    """from_iso + 7 years, ISO string. Used for incident soft-delete (regulatory clock)."""
    73	    dt = datetime.fromisoformat(from_iso.replace("Z", "+00:00"))
    74	    return (dt + timedelta(days=365 * INCIDENT_RETENTION_YEARS)).isoformat()

    77	async def _ensure_company_or_placeholder(
    78	    db, project_id: str, company_id: Optional[str], company_name: Optional[str], actor_id: str
    79	) -> Optional[str]:
    80	    """
    81	    If company_id is provided and exists on this project (non-deleted) — return it.
    82	    If only company_name is provided — create or reuse a placeholder project_companies doc.
    83	    If neither — return None.
    84	    """
    85	    if company_id:
    86	        doc = await db.project_companies.find_one({
    87	            "id": company_id,
    88	            "project_id": project_id,
    89	            "deletedAt": None,
    90	        })
    91	        if doc:
    92	            return company_id
    93	        raise HTTPException(status_code=404, detail="company_id not found on this project")
    94	
    95	    if not company_name:
    96	        return None
    97	
    98	    name = company_name.strip()
    99	    existing = await db.project_companies.find_one({
   100	        "project_id": project_id,
   101	        "name": name,
   102	        "deletedAt": None,
   103	    })
   104	    if existing:
   105	        return existing["id"]
   106	
   107	    new_id = _new_id()
   108	    await db.project_companies.insert_one({
   109	        "id": new_id,
   110	        "project_id": project_id,
   111	        "name": name,
   112	        "is_placeholder": True,
   113	        "created_at": _now(),
   114	        "created_by": actor_id,
   115	        "deletedAt": None,
   116	        "deletedBy": None,
   117	    })
   118	    await _audit("project_company", new_id, "create_placeholder", actor_id, {
   119	        "project_id": project_id, "name": name, "source": "safety",
   120	    })
   121	    return new_id

   729	
   730	def _assert_transition_allowed(
   731	    current: str, next_status: str, project_role: str, payload_dict: dict
   732	) -> None:
   733	    """Enforce status transition rules. Raises 409 on invalid transition."""
   734	    if current == next_status:
   735	        return  # idempotent
   736	    # Spec: open → in_progress → completed; any → cancelled; reverse = 409
   737	    allowed = {
   738	        "open": {"in_progress", "cancelled"},
   739	        "in_progress": {"completed", "cancelled"},
   740	        "completed": {"cancelled"},
   741	        "cancelled": set(),
   742	    }
   743	    if next_status not in allowed.get(current, set()):
   744	        raise HTTPException(status_code=409, detail={
   745	            "current_status": current,
   746	            "requested": next_status,
   747	            "reason": "transition not allowed",
   748	        })
   749	    if next_status == "completed":
   750	        ca = payload_dict.get("corrective_action")
   751	        if not ca or not str(ca).strip():
   752	            raise HTTPException(status_code=409, detail={
   753	                "current_status": current,
   754	                "requested": next_status,
   755	                "reason": "corrective_action is required to mark a task completed",
   756	            })
   757	    if next_status == "cancelled":
   758	        if project_role != "project_manager":
   759	            raise HTTPException(status_code=409, detail={
   760	                "current_status": current,
   761	                "requested": next_status,
   762	                "reason": "only project_manager may cancel a task",
   763	            })
