Why v2 is safer:

  We are literally cloning a pattern that already runs in staging today.

  already registered.

- Embedded data = atomic upsert. No multi-document race conditions.
- Rollback is trivial: revert removes the field; existing data is unaffected
  (MongoDB ignores unknown fields).

- PDF integration (Section 4 of the regulatory PDF) — S2B v3.
- Sync with project memberships (autocomplete from existing team) — Phase 2.
- Linkage between personnel and incidents/tours — Phase 2.

   FULL FILE CONTENTS:

       /**
        * Safety Project Personnel Section — Batch S2B v2 (2026-05-03).
        *
        * Renders a collapsible section grouped by 5 safety roles. Each
        * role has its own "+ הוסף" button and inline list of assignees.
        * Multiple personnel per role allowed (no count limit).
        *
        * Data flow: fully controlled by parent — receives `personnel`
        * array + `onChange` callback. No internal fetch / save. Saves
        * happen via the parent's upsertRegistration flow.
        *
        * Roles match VALID_SUB_ROLES (invites_router.py:145) — document
        * labels only, NOT permissions.
        *
        * Capacitor: all touch targets py-3 (≥44px). ID inputs type="text"
        * to preserve leading zeros. PII id_number is masked by server on
        * save (1***789); raw value never persisted.
        */
       import React, { useState } from 'react';
       import { Plus, Trash2, ChevronDown, ChevronUp } from 'lucide-react';

       const ROLE_LABELS = {
         site_manager:       'מנהל אתר',
         execution_engineer: 'מהנדס ביצוע',
         safety_assistant:   'עוזר בטיחות',
         work_manager:       'מנהל עבודה',
         safety_officer:     'ממונה בטיחות',
       };

       const ROLE_ORDER = [
         'site_manager',
         'execution_engineer',
         'safety_assistant',
         'work_manager',
         'safety_officer',
       ];

       const EMPTY_PERSON = (role) => ({
         role,
         first_name: '',
         last_name: '',
         id_number: '',
         license_number: '',
         notes: '',
       });

       export default function SafetyProjectPersonnelSection({
         personnel = [],
         onChange,
         isOpen,
         onToggle,
       }) {
         // Group personnel by role for display
         const grouped = ROLE_ORDER.reduce((acc, role) => {
           acc[role] = personnel
             .map((p, i) => ({ ...p, _idx: i }))
             .filter(p => p.role === role);
           return acc;
         }, {});

         const totalCount = personnel.length;

         const addPerson = (role) => {
           onChange([...personnel, EMPTY_PERSON(role)]);
         };

         const removePerson = (idx) => {
           if (!window.confirm('להסיר את האחראי הזה?')) return;
           const next = personnel.filter((_, i) => i !== idx);
           onChange(next);
         };

         const updatePerson = (idx, field, value) => {
           const next = personnel.map((p, i) => (
             i === idx ? { ...p, [field]: value } : p
           ));
           onChange(next);
         };

         return (
           <div className="border border-slate-200 rounded-lg bg-white">
             {/* Header — collapsible */}
             <button
               type="button"
               onClick={onToggle}
               className="w-full flex items-center justify-between py-3 px-4 text-right"
             >
               <span className="text-slate-500">
                 {isOpen ? <ChevronUp className="w-5 h-5" /> : <ChevronDown className="w-5 h-5" />}
               </span>
               <span className="font-semibold text-slate-800">
                 4. אחראי בטיחות בפרויקט ({totalCount})
               </span>
             </button>

             {isOpen && (
               <div className="px-4 pb-4 space-y-4">
                 {ROLE_ORDER.map(role => (
                   <div
                     key={role}
                     className="border-t border-slate-100 pt-3 first:border-t-0 first:pt-0"
                   >
                     <div className="flex items-center justify-between mb-2">
                       <span className="text-xs text-slate-500">
                         ({grouped[role].length})
                       </span>
                       <h4 className="font-medium text-slate-700">{ROLE_LABELS[role]}</h4>
                     </div>

                     {/* Existing personnel for this role */}
                     {grouped[role].length === 0 && (
                       <p className="text-xs text-slate-400 mb-2">לא הוגדרו {ROLE_LABELS[role]}</p>
                     )}

                     <div className="space-y-3 mb-2">
                       {grouped[role].map(person => (
                         <div
                           key={person._idx}
                           className="border border-slate-200 rounded p-3 bg-slate-50 space-y-2"
                         >
                           <div className="flex items-start justify-between">
                             <button
                               type="button"
                               onClick={() => removePerson(person._idx)}
                               className="text-red-500 hover:text-red-700 p-2"
                               aria-label="הסר"
                             >
                               <Trash2 className="w-4 h-4" />
                             </button>
                             <span className="text-xs text-slate-500">
                               #{grouped[role].indexOf(person) + 1}
                             </span>
                           </div>
                           <div className="grid grid-cols-2 gap-2">
                             <input
                               type="text"
                               placeholder="שם פרטי *"
                               value={person.first_name}
                               onChange={(e) => updatePerson(person._idx, 'first_name', e.target.value)}
                               className="border border-slate-300 rounded px-3 py-3 text-right text-sm bg-white"
                             />
                             <input
                               type="text"
                               placeholder="שם משפחה *"
                               value={person.last_name}
                               onChange={(e) => updatePerson(person._idx, 'last_name', e.target.value)}
                               className="border border-slate-300 rounded px-3 py-3 text-right text-sm bg-white"
                             />
                           </div>
                           <input
                             type="text"
                             placeholder="ת.ז (אופציונלי)"
                             value={person.id_number}
                             onChange={(e) => updatePerson(person._idx, 'id_number', e.target.value)}
                             className="w-full border border-slate-300 rounded px-3 py-3 text-right text-sm bg-white"
                           />
                           <input
                             type="text"
                             placeholder="מספר רישיון בטיחות (אופציונלי)"
                             value={person.license_number}
                             onChange={(e) => updatePerson(person._idx, 'license_number', e.target.value)}
                             className="w-full border border-slate-300 rounded px-3 py-3 text-right text-sm bg-white"
                           />
                           <textarea
                             placeholder="הערות"
                             value={person.notes}
                             onChange={(e) => updatePerson(person._idx, 'notes', e.target.value)}
                             rows={2}
                             className="w-full border border-slate-300 rounded px-3 py-3 text-right text-sm bg-white"
                           />
                         </div>
                       ))}
                     </div>

                     <button
                       type="button"
                       onClick={() => addPerson(role)}
                       className="w-full flex items-center justify-center gap-2 py-3 px-4 border border-dashed border-slate-300 rounded text-sm text-slate-600 hover:bg-slate-50"
                     >
                       <Plus className="w-4 h-4" /> הוסף {ROLE_LABELS[role]}
                     </button>
                   </div>
                 ))}
               </div>
             )}
           </div>
         );
       }

NOTES:
- All inputs use py-3 → ≥44px touch target (CLAUDE.md Capacitor rule).
- ID input is type="text" (not "number") to preserve leading zeros.
- The `_idx` injected via map+filter is a positional index in the parent's
  full array — used for update/remove to mutate the right entry across
  role groups.
- No internal fetch/save — fully controlled by parent. This is intentional
  and matches the managers section pattern from #468.
