  import React, { useCallback } from 'react';
  import { Camera } from 'lucide-react';
  import { DocumentScanner } from '@capacitor-mlkit/document-scanner';
  import { Capacitor } from '@capacitor/core';
  import { toast } from 'sonner';

  /**
   * BATCH H.2-Bundle (2026-05-13) — wrapper around the native
   * document scanner. iOS uses VNDocumentCameraViewController
   * (Apple system); Android uses Google ML Kit Document Scanner.
   * Falls back to nothing (button hidden) on web.
   *
   * Props:
   *   - onScan: (files: File[]) => void — called with the
   *     scanned pages as File objects (converted from native URIs).
   *   - label: string — button text (default "סרוק מסמך")
   *   - className: string — wrapper styling
   */
  export default function DocumentScannerButton({ onScan, label = 'סרוק מסמך', className = '' }) {
    const handleClick = useCallback(async () => {
      if (!Capacitor.isNativePlatform()) {
        toast.error('סריקת מסמכים זמינה רק באפליקציה');
        return;
      }
      try {
        const result = await DocumentScanner.scan();
        const scannedFiles = await Promise.all(
          (result.scannedImages || []).map(async (uri, i) => {
            const resp = await fetch(uri);
            const blob = await resp.blob();
            return new File([blob], `scan-${Date.now()}-${i}.jpg`, { type: 'image/jpeg' });
          })
        );
        if (scannedFiles.length > 0) onScan(scannedFiles);
      } catch (err) {
        if (err.message?.includes('cancel')) return; // user dismissed
        console.error('DocumentScanner error:', err);
        toast.error('שגיאה בסריקה');
      }
    }, [onScan]);

    if (!Capacitor.isNativePlatform()) return null;

    return (
      <button onClick={handleClick} className={className}>
        <Camera className="w-4 h-4" />
        {label}
      </button>
    );
  }

LOCATION 2.5 — Add appUrlOpen listener via <DeepLinkHandler> component
File: frontend/src/App.js

PATTERN (verified by STEP 0.13): App uses `BrowserRouter` (react-
router-dom v6). The `useNavigate` hook only works in components
INSIDE BrowserRouter. The appUrlOpen listener must therefore live
inside a component that's rendered INSIDE BrowserRouter — NOT in
the top-level App component scope.

**Solution:** create a tiny `<DeepLinkHandler>` component that
renders nothing but uses `useNavigate` + registers the listener in
a useEffect. Mount it INSIDE BrowserRouter.

ADD (near top of file, after the existing helper components like
`NetworkPlaceholder`):

  // BATCH H.2-Bundle (2026-05-13) — Universal Links + App Links.
  // Lives INSIDE BrowserRouter so it can call useNavigate. When the
  // OS hands off https://app.brikops.com/<path> to the app (user
  // tapped a link from WhatsApp/SMS/email + app is installed),
  // parses the path and navigates via React Router. If app NOT
  // installed, OS falls back to browser automatically.
  //
  // Auth-bounce: protected routes already redirect logged-out
  // users to /login while storing INTENDED_PATH_KEY +
  // deepLinkSource='external'. After login, the existing
  // LoginPage/WaLoginPage/OnboardingPage logic navigates back to
  // the intended path. No new code needed for that flow.
  const DeepLinkHandler = () => {
    const navigate = useNavigate();

    useEffect(() => {
      const handler = ({ url }) => {
        try {
          const u = new URL(url);
          // Defense in depth — only handle our own domain.
          // (autoVerify should prevent foreign URLs, but cheap
          // insurance against future intent-filter conflicts.)
          if (u.hostname !== 'app.brikops.com') return;
          const path = u.pathname + (u.search || '') + (u.hash || '');
          if (!path || path === '/') return;
          // Mark this navigation as external (used by some pages
          // like TaskDetailPage:224 to distinguish in-app vs
          // external entry).
          try { sessionStorage.setItem('deepLinkSource', 'external'); } catch {}
          navigate(path);
        } catch (err) {
          console.warn('appUrlOpen parse failed:', err);
        }
      };

      let listenerHandle;
      CapacitorApp.addListener('appUrlOpen', handler).then(h => {
        listenerHandle = h;
      });

      return () => {
        if (listenerHandle?.remove) listenerHandle.remove();
      };
    }, [navigate]);

    return null;
  };

  import React, { useCallback } from 'react';
  import { Users } from 'lucide-react';
  import { Contacts } from '@capacitor-community/contacts';
  import { Capacitor } from '@capacitor/core';
  import { toast } from 'sonner';

  /**
   * BATCH H.2-Bundle (2026-05-13) — wrapper around the native
   * contact picker. iOS uses CNContactPickerViewController;
   * Android uses ContactsContract picker intent. App never sees
   * the full address book — only the contact the user picks.
   *
   * Props:
   *   - onPicked: ({ name: string, phone: string }) => void
   *   - label: string — button text (default "בחר מאנשי קשר")
   *   - className: string — wrapper styling
   */
  export default function ContactPickerButton({ onPicked, label = 'בחר מאנשי קשר', className = '' }) {
    const handleClick = useCallback(async () => {
      if (!Capacitor.isNativePlatform()) return; // hidden on web
      try {
        // Request permission first (Capacitor plugin auto-handles).
        const perm = await Contacts.requestPermissions();
        if (perm.contacts !== 'granted') {
          // BATCH H.2-Bundle — tell user HOW to fix, not just THAT
          // it's broken. After first denial, iOS won't re-prompt
          // — user must enable manually in Settings. Apple
          // reviewers check this edge case.
          toast.error(
            'נדרשת הרשאה לאנשי קשר. ' +
            'ניתן להפעיל בהגדרות > BrikOps > אנשי קשר.'
          );
          return;
        }
        const { contact } = await Contacts.pickContact({
          projection: { name: true, phones: true }
        });
        if (!contact) return; // user cancelled
        const name = contact.name?.display || '';
        const phone = contact.phones?.[0]?.number || '';
        onPicked({ name, phone });
      } catch (err) {
        if (err.message?.includes('cancel')) return;
        console.error('ContactPicker error:', err);
        toast.error('שגיאה בבחירת איש קשר');
      }
    }, [onPicked]);

    if (!Capacitor.isNativePlatform()) return null;

    return (
      <button onClick={handleClick} className={className}>
        <Users className="w-4 h-4" />
        {label}
      </button>
    );
  }

RISK

Medium. Native batch with Apple/Android review cycle. First native
release in a while = some accumulated drift risk.

  webcredentials block is dropped, Apple Sign-In stops working
  (Pre-flight #3 + #5 guard this).

  cert. Wrong fingerprint → Android falls back to browser
  (graceful but not desired). Pre-flight verify before ship.
- ⚠ Apple reviewer reads NSContactsUsageDescription Hebrew text.
  Drafted text emphasizes "we don't send to our servers" —
  privacy-clear. If reviewer rejects, follow-up wording tweak.
- ⚠ `appUrlOpen` listener — implementation note about navigate
  ref placement. Replit may need to choose the cleanest React
  Router pattern matching existing code style.
- ⚠ Document scanner ML Kit on Android downloads model on first
  use (~1-2MB). First scan may have brief delay.
- ⚠ Contact picker plugin returns ONE contact (not list). Apple
  reviewer expectation; matches privacy commitment in Hebrew
  rationale.
- ⚠ Universal Links require the AASA to be served on the SIGNED
  prod domain (app.brikops.com). Staging won't auto-resolve;
  test via prod with demo accounts.
- ⚠ Bundle size: +2 Capacitor plugins. Estimate: ~500KB iOS, ~1MB
  Android (ML Kit model gets downloaded post-install on first use).
  Acceptable.
- ⚠ Apple review window: 1-3 days expected. Plan accordingly.
- ⚠ Backend / DB / env / payment / billing: zero touch.
