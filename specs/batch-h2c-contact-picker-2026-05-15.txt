================================================================================
Date:           2026-05-15
Branch:         staging (NOT main)
Baseline:       HEAD post-H.2b + hotfix-1 + Android version bump
                (Universal Links confirmed working on iOS prod)
Status:         AWAITING REPLIT IMPLEMENTATION

What this enables:
  When a user (PM, contractor manager) opens an invite-by-phone form,
  a small "📇 בחר מאנשי קשר" button appears next to the phone input
  (NATIVE ONLY — invisible on web). Tapping it opens the iOS / Android
  system contact picker. After the user picks a contact, the **phone
  number** is auto-filled into the form. The name is NOT auto-filled —
  the inviter still types it manually on the same screen.

Why phone-only autofill (Zahi's product decision, 2026-05-15):
  The inviter is naming the person by their role in the project
  (e.g., "אבי הקבלן" not the system contact name "Avi Cohen").
  The phone number is the unique identifier — that's what the
  invitation backend uses (OTP, WhatsApp links). Pulling only the
  phone keeps the form fast without overriding contextual naming.

  // BATCH H.2c (2026-05-15) — Contact Picker
  // ----------------------------------------------------------------
  // Wraps @capacitor-community/contacts. Native-only.
  // On web → returns null (no button rendered).
  // On native → renders icon button. On press:
  //   1. Request permission if not granted.
  //   2. If denied → show toast "ניתן להפעיל ב-Settings".
  //   3. If granted → open native contact picker.
  //   4. On contact picked → extract first phone number,
  //      normalize, call onPhonePicked(normalizedPhone).
  //
  // Phone normalization:
  //   - Strip all non-digit chars except leading +
  //   - If starts with +972 → replace with 0
  //   - If starts with 972 (no +) → replace with 0
  //   - Result: "0501234567" style
  //
  // Props:
  //   onPhonePicked: (phone: string) => void   (required)
  //   disabled?: boolean                        (default false)
  //   className?: string                        (default '')

  import { useState } from 'react';
  import { Capacitor } from '@capacitor/core';
  import { Contacts } from '@capacitor-community/contacts';

  function normalizePhone(raw) {
    if (!raw) return '';
    let p = String(raw).replace(/[^\d+]/g, '');
    if (p.startsWith('+972')) p = '0' + p.slice(4);
    else if (p.startsWith('972')) p = '0' + p.slice(3);
    else if (p.startsWith('+')) p = p.slice(1);     // foreign — strip + only
    return p;
  }

  export default function ContactPickerButton({ onPhonePicked, disabled = false, className = '' }) {
    const [busy, setBusy] = useState(false);

    if (!Capacitor.isNativePlatform()) return null;   // web fallback: invisible

    const handlePick = async () => {
      if (disabled || busy) return;
      setBusy(true);
      try {
        const perm = await Contacts.requestPermissions();
        if (perm.contacts !== 'granted') {
          // soft fail — user can still type manually
          console.warn('Contacts permission denied');
          // TODO: show toast (use existing toast helper if present, else console)
          return;
        }
        const result = await Contacts.pickContact({
          projection: { phones: true, name: true }   // phone-only autofill, name not used
        });
        const phones = result?.contact?.phones || [];
        if (phones.length === 0) {
          console.warn('Picked contact has no phone');
          return;
        }
        const phone = normalizePhone(phones[0].number);
        if (phone) onPhonePicked(phone);
      } catch (err) {
        console.warn('Contact picker error:', err);
      } finally {
        setBusy(false);
      }
    };

    return (
      <button
        type="button"
        onClick={handlePick}
        disabled={disabled || busy}
        className={`px-3 py-2 text-sm border border-slate-300 rounded-lg bg-white hover:bg-slate-50 disabled:opacity-50 ${className}`}
        aria-label="בחר מאנשי קשר"
      >
        📇 בחר מאנשי קשר
      </button>
    );
  }
