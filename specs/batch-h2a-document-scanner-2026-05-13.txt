LOCATION 3 — NEW: DocumentScannerButton.jsx
================================================================================
File: frontend/src/components/DocumentScannerButton.jsx (NEW)

  import React, { useCallback } from 'react';
  import { Camera } from 'lucide-react';
  import { DocumentScanner } from '@capacitor-mlkit/document-scanner';
  import { Capacitor } from '@capacitor/core';
  import { toast } from 'sonner';

  /**
   * BATCH H.2a (2026-05-13) — wrapper around the native document
   * scanner. iOS uses VNDocumentCameraViewController (Apple system
   * Vision Kit). Android uses Google ML Kit Document Scanner.
   *
   * Hidden on web (no plugin functionality there).
   *
   * Props:
   *   - onScan: (files: File[]) => void
   *     Called with the scanned pages as File objects (JPG, one per
   *     page). Consumer pushes into images[] via existing flow.
   *   - label?: string (default "סרוק מסמך")
   *   - className?: string (styling — match the existing
   *     "צלם תמונה" button)
   */
  export default function DocumentScannerButton({
    onScan,
    label = 'סרוק מסמך',
    className = ''
  }) {
    const handleClick = useCallback(async () => {
      if (!Capacitor.isNativePlatform()) {
        toast.error('סריקת מסמכים זמינה רק באפליקציה');
        return;
      }

      try {
        const result = await DocumentScanner.scan();
        const uris = result?.scannedImages || [];

        if (uris.length === 0) return; // user cancelled

        // Convert each URI to a File so the existing photo flow can
        // consume it. Mirrors how a regular camera shot becomes a File.
        const files = await Promise.all(uris.map(async (uri, i) => {
          const resp = await fetch(uri);
          const blob = await resp.blob();
          return new File(
            [blob],
            `scan-${Date.now()}-${i}.jpg`,
            { type: 'image/jpeg' }
          );
        }));

        if (files.length > 0) onScan(files);
      } catch (err) {
        // User cancellation tends to throw — silence those.
        if (err?.message?.toLowerCase?.().includes('cancel')) return;
        console.error('DocumentScanner error:', err);
        toast.error('שגיאה בסריקת המסמך');
      }
    }, [onScan]);

    // Hide on web — there's no native scanner there.
    if (!Capacitor.isNativePlatform()) return null;

    return (
      <button
        type="button"
        onClick={handleClick}
        className={className}
      >
        <Camera className="w-4 h-4 ml-1" />
        {label}
      </button>
    );
  }

Notes:
- `Capacitor.isNativePlatform()` returns false on web → button hides
  → no broken UX in the web build.
- Errors from user cancellation are silenced. Other errors get a
  Hebrew toast.
- File naming pattern (`scan-{timestamp}-{index}.jpg`) mirrors the
  existing convention for annotated photos (e.g., `annotated.jpg`).
