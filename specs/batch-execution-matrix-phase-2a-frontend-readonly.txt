ARCHITECTURAL DECISIONS
================================================================================

D1. View mode by viewport (auto), not user choice in 2A
    - Mobile (<768px): MatrixListView
    - Desktop (≥768px): MatrixGridView
    - Manual toggle DEFERRED to Phase 2D (some power users on tablet
      may prefer list mode — add toggle then if requested)

D2. Status icons = SVG (Lucide), NOT emoji
    Per CLAUDE.md: "use SVG icons (Heroicons, Lucide), not emojis".
    Map of 6 status → Lucide icon below in STATUS_CONFIG.

D3. Touch targets ≥44px
    Each MatrixCell wrapped in py-3 px-3 button on mobile = 44px+ height.
    On desktop grid, cells are 40×40px but inside a clickable div with
    padding that totals 44px+ (mouse hover area can be smaller).

D4. RTL throughout
    All components dir="rtl" or inherit. Stage names + unit numbers
    align right. Horizontal scroll direction respects RTL (right-to-left
    primary direction; React's overflow-x-auto handles automatically).

D5. No realtime updates in 2A
    Page fetches once on mount. User refreshes manually if needed.
    Phase 2B will refresh on optimistic cell edit, Phase 3 may add WebSocket.

D6. Loading + error + empty states
    - Loading: skeleton screens (gray bars sized to expected content)
    - Error: red Alert with retry button
    - Empty (no units in project): "אין דירות בפרויקט" + link to project setup

D7. Service layer = matrixService.js (mirrors qcService pattern)
    Single responsibility: HTTP requests against /api/execution-matrix.
    Hook layer = useMatrixData.js for data fetching + state mgmt.
    Component layer = pure rendering, no API calls.

       import { useState, useEffect, useCallback, useRef } from 'react';
       import { matrixService } from '../services/matrixService';

       /**
        * Custom hook for fetching + managing matrix data.
        * Handles loading, error, abort-on-unmount.
        */
       export function useMatrixData(projectId) {
         const [data, setData] = useState(null);
         const [loading, setLoading] = useState(true);
         const [error, setError] = useState(null);
         const abortRef = useRef(null);

         const load = useCallback(async () => {
           if (!projectId) return;
           if (abortRef.current) abortRef.current.abort();
           const controller = new AbortController();
           abortRef.current = controller;
           setLoading(true);
           setError(null);
           try {
             const result = await matrixService.getMatrix(projectId);
             if (!controller.signal.aborted) {
               setData(result);
               setLoading(false);
             }
           } catch (err) {
             if (!controller.signal.aborted) {
               setError(err?.response?.data?.detail || 'שגיאה בטעינת המטריצה');
               setLoading(false);
             }
           }
         }, [projectId]);

         useEffect(() => {
           load();
           return () => {
             if (abortRef.current) abortRef.current.abort();
           };
         }, [load]);

         return { data, loading, error, refresh: load };
       }

       import React from 'react';
       import { MATRIX_STATUSES, EMPTY_CELL_CONFIG } from './STATUS_CONFIG';

       /**
        * Single matrix cell — renders status icon + bg color.
        * Phase 2A: read-only (no onClick).
        * Phase 2B: will add onClick → CellPopover.
        *
        * Props:
        *   cell: { status, text_value, note, last_actor_name } | null
        *   stage: { id, type } — for tag-type stages, render text_value
        *   size: 'sm' (32px, mobile list view) | 'md' (40px, desktop grid)
        */
       export default function MatrixCell({ cell, stage, size = 'sm' }) {
         const isTag = stage?.type === 'tag';
         const status = cell?.status;
         const textValue = cell?.text_value;

         // Tag column — show text snippet
         if (isTag) {
           return (
             <div
               className={`flex items-center justify-center rounded-md border ${
                 textValue
                   ? 'bg-violet-50 text-violet-700 border-violet-200'
                   : `${EMPTY_CELL_CONFIG.bg} ${EMPTY_CELL_CONFIG.text} ${EMPTY_CELL_CONFIG.border}`
               } ${size === 'sm' ? 'h-8 w-8' : 'h-10 w-10'}`}
               title={textValue || 'ללא תגית'}
               dir="rtl"
             >
               <span className="text-[10px] font-medium truncate px-1">
                 {textValue ? textValue.slice(0, 3) : '—'}
               </span>
             </div>
           );
         }

         // Status column — show status icon
         const cfg = status ? MATRIX_STATUSES[status] : null;
         if (cfg) {
           const Icon = cfg.Icon;
           return (
             <div
               className={`flex items-center justify-center rounded-md border ${cfg.bg} ${cfg.text} ${cfg.border} ${
                 size === 'sm' ? 'h-8 w-8' : 'h-10 w-10'
               }`}
               title={cfg.label + (cell?.last_actor_name ? ` • ${cell.last_actor_name}` : '')}
               dir="rtl"
             >
               <Icon className={size === 'sm' ? 'w-4 h-4' : 'w-5 h-5'} />
             </div>
           );
         }

         // Empty cell
         return (
           <div
             className={`flex items-center justify-center rounded-md border ${EMPTY_CELL_CONFIG.bg} ${EMPTY_CELL_CONFIG.text} ${EMPTY_CELL_CONFIG.border} ${
               size === 'sm' ? 'h-8 w-8' : 'h-10 w-10'
             }`}
             title="לא סומן"
             dir="rtl"
           >
             <span className="text-xs">—</span>
           </div>
         );
       }

   FULL FILE:

       import React, { useMemo } from 'react';
       import MatrixCell from './MatrixCell';

       /**
        * Mobile-first list view: each unit = card, stages scroll horizontally.
        *
        * Props:
        *   units: [{ id, unit_no, floor_id, room_count, ... }]
        *   stages: [{ id, title, type, source, scope }]
        *   cells: [{ unit_id, stage_id, status, text_value, note, last_actor_name }]
        *   floorsById: { [floor_id]: { id, floor_number, name } }
        */
       export default function MatrixListView({ units, stages, cells, floorsById }) {
         // Index cells by unit_id + stage_id for O(1) lookup
         const cellsByUnitStage = useMemo(() => {
           const map = {};
           for (const c of cells) {
             const key = `${c.unit_id}::${c.stage_id}`;
             map[key] = c;
           }
           return map;
         }, [cells]);

         // Per-unit progress: count completed status / total stages
         const progressForUnit = (unitId) => {
           let completed = 0;
           let total = 0;
           for (const stage of stages) {
             if (stage.type !== 'status') continue;  // tag columns don't count
             total += 1;
             const c = cellsByUnitStage[`${unitId}::${stage.id}`];
             if (c?.status === 'completed' || c?.status === 'no_findings') {
               completed += 1;
             }
           }
           return { completed, total, pct: total ? Math.round((completed / total) * 100) : 0 };
         };

         if (!units || units.length === 0) {
           return (
             <div className="text-center py-12 text-slate-500" dir="rtl">
               אין דירות בפרויקט
             </div>
           );
         }

         return (
           <div className="space-y-3" dir="rtl">
             {units.map((unit) => {
               const floor = floorsById[unit.floor_id];
               const { completed, total, pct } = progressForUnit(unit.id);
               return (
                 <div
                   key={unit.id}
                   className="bg-white rounded-xl border border-slate-200 p-3"
                 >
                   {/* Unit header */}
                   <div className="flex items-center justify-between mb-2">
                     <div className="flex items-center gap-2 min-w-0">
                       <span className="text-sm font-bold text-slate-900 shrink-0">
                         דירה {unit.unit_no}
                       </span>
                       {floor && (
                         <span className="text-xs text-slate-500 truncate">
                           • קומה {floor.floor_number}
                           {unit.room_count != null && ` • ${unit.room_count} ח'`}
                         </span>
                       )}
                     </div>
                     <span className="text-xs font-medium text-slate-600 shrink-0">
                       {completed}/{total}
                     </span>
                   </div>

                   {/* Progress bar */}
                   <div className="h-1.5 bg-slate-100 rounded-full overflow-hidden mb-3">
                     <div
                       className="h-full bg-emerald-500 transition-all"
                       style={{ width: `${pct}%` }}
                     />
                   </div>

                   {/* Horizontal scroll of stages */}
                   <div className="flex gap-1.5 overflow-x-auto pb-1 -mx-1 px-1">
                     {stages.map((stage) => {
                       const cell = cellsByUnitStage[`${unit.id}::${stage.id}`];
                       return (
                         <div key={stage.id} className="flex flex-col items-center shrink-0 w-12">
                           <MatrixCell cell={cell} stage={stage} size="sm" />
                           <span className="text-[9px] text-slate-500 mt-1 text-center leading-tight w-full truncate" title={stage.title}>
                             {stage.title}
                           </span>
                         </div>
                       );
                     })}
                   </div>
                 </div>
               );
             })}
           </div>
         );
       }

NOTES:
- Each unit card has all stages, scrollable horizontally.
- Progress bar shows completed/total only counting STATUS columns
  (tag columns excluded from progress math).
- 12px (`w-12`) per stage cell + label = ~80 stages fit in scroll on
  smartphone. Real projects have ≤30 stages — fits well.
- `room_count` shown when available (some units don't have it).

           }
           return map;

           return (
             <div className="text-center py-12 text-slate-500" dir="rtl">
               אין דירות בפרויקט
             </div>
           );

               <thead className="sticky top-0 z-10 bg-slate-50 border-b border-slate-200">
                 <tr>
                   <th className="sticky right-0 z-20 bg-slate-50 border-l border-slate-200 px-3 py-2 text-right text-xs font-medium text-slate-700 min-w-[120px]">

                       title={stage.title}
                     >
                       <div className="truncate max-w-[80px] mx-auto">{stage.title}</div>
                       {stage.source === 'custom' && (
                         <div className="text-[9px] text-violet-600 mt-0.5">מותאם</div>
                       )}
                     </th>
                   ))}
                 </tr>
               </thead>
               <tbody>
                 {units.map((unit, idx) => {
                   const floor = floorsById[unit.floor_id];
                   return (
                     <tr key={unit.id} className={idx % 2 ? 'bg-slate-50/30' : 'bg-white'}>
                       <td className="sticky right-0 z-10 bg-inherit border-l border-slate-200 px-3 py-2 text-right text-xs">
                         <div className="font-medium text-slate-900">דירה {unit.unit_no}</div>

                           <div className="text-[10px] text-slate-500 mt-0.5">
                             קומה {floor.floor_number}
                           </div>
                         )}
                       </td>
                       {stages.map((stage) => {
                         const cell = cellsByUnitStage[`${unit.id}::${stage.id}`];
                         return (
                           <td key={stage.id} className="p-1.5 border-l border-slate-100 last:border-l-0 text-center">
                             <div className="inline-flex">
                               <MatrixCell cell={cell} stage={stage} size="md" />
                             </div>
                           </td>
                         );
                       })}
                     </tr>
                   );
                 })}
               </tbody>
             </table>
           </div>
         );
       }

                       </button>
                     </div>
                   </div>
                 </div>
